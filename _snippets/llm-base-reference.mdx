<ResponseField name="model" type="str">
  ID of the model to use.
</ResponseField>
<ResponseField name="name" type="str">
  Name for this LLM. Note: This is not sent to the LLM API.
</ResponseField>
<ResponseField name="metrics" type="Dict[str, Any]">
  Metrics collected for this LLM. Note: This is not sent to the LLM API.
</ResponseField>
<ResponseField name="response_format" type="Any">
  Format of the response.
</ResponseField>
<ResponseField name="tools" type="List[Union[Tool, Dict]]">
  A list of tools provided to the LLM.
  Tools are functions the model may generate JSON inputs for.
  If you provide a dict, it is not called by the model.
  Always add tools using the add_tool() method.
</ResponseField>
<ResponseField name="tool_choice" type="Union[str, Dict[str, Any]]">
  Controls which (if any) function is called by the model.
  "none" means the model will not call a function and instead generates a message.
  "auto" means the model can pick between generating a message or calling a function.
  Specifying a particular function via \{"type: "function", "function": \{"name": "my_function"}}
  forces the model to call that function.
  "none" is the default when no functions are present. "auto" is the default if functions are present.
</ResponseField>
<ResponseField name="run_tools" type="bool" default="True">
  If True, runs tools.
</ResponseField>
<ResponseField name="show_tool_calls" type="bool">
  If True, shows tool calls in the response.
</ResponseField>
<ResponseField name="functions" type="Dict[str, Function]">
  Functions extracted from the tools. Note: These are not sent to the LLM API and are only used for execution.
</ResponseField>
<ResponseField name="function_call_limit" type="int" default="20">
  Maximum number of function calls allowed.
</ResponseField>
<ResponseField name="function_call_stack" type="List[FunctionCall]">
  Stack of function calls.
</ResponseField>
<ResponseField name="system_prompt" type="str">
  System prompt provided to the LLM.
</ResponseField>
<ResponseField name="instructions" type="str">
  Instructions provided to the LLM.
</ResponseField>
