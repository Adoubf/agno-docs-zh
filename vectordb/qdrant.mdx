## Example

```python agent_with_knowledge.py
import os
import typer
from typing import Optional
from rich.prompt import Prompt

from phi.agent import Agent
from phi.knowledge.pdf import PDFUrlKnowledgeBase
from phi.vectordb.qdrant import Qdrant

api_key = os.getenv("QDRANT_API_KEY")
qdrant_url = os.getenv("QDRANT_URL")
collection_name = "thai-recipe-index"

vector_db = Qdrant(
    table_name=collection_name,
    url=qdrant_url,
    api_key=api_key,
)

knowledge_base = PDFUrlKnowledgeBase(
    urls=["https://phi-public.s3.amazonaws.com/recipes/ThaiRecipes.pdf"],
    vector_db=vector_db,
)

# Comment out after first run
knowledge_base.load(recreate=True, upsert=True)


def qdrant_agent(user: str = "user"):
    run_id: Optional[str] = None

    agent = Agent(
        run_id=run_id,
        user_id=user,
        knowledge=knowledge_base,
        tool_calls=True,
        use_tools=True,
        show_tool_calls=True,
        debug_mode=True,
    )

    if run_id is None:
        run_id = agent.run_id
        print(f"Started Run: {run_id}\n")
    else:
        print(f"Continuing Run: {run_id}\n")

    while True:
        message = Prompt.ask(f"[bold] :sunglasses: {user} [/bold]")
        if message in ("exit", "bye"):
            break
        agent.print_response(message)


if __name__ == "__main__":
    typer.run(qdrant_agent)

```

## Qdrant Params

<ResponseField name="collection" type="str">
  The name of the collection to use.
</ResponseField>
<ResponseField name="embedder" type="Embedder">
  The embedder to use.
</ResponseField>
<ResponseField name="distance" type="Distance">
  The distance to use.
</ResponseField>
<ResponseField name="location" type="Optional[str]">
  The location of the Qdrant database.
</ResponseField>
<ResponseField name="url" type="Optional[str]">
  The URL of the Qdrant server.
</ResponseField>
<ResponseField name="port" type="Optional[int]">
  The port number for the Qdrant server. Default is 6333.
</ResponseField>
<ResponseField name="grpc_port" type="int">
  The gRPC port number. Default is 6334.
</ResponseField>
<ResponseField name="prefer_grpc" type="bool">
  Whether to prefer gRPC over HTTP. Default is False.
</ResponseField>
<ResponseField name="https" type="Optional[bool]">
  Whether to use HTTPS for connection.
</ResponseField>
<ResponseField name="api_key" type="Optional[str]">
  The API key for authentication.
</ResponseField>
<ResponseField name="prefix" type="Optional[str]">
  The prefix to use for the Qdrant client.
</ResponseField>
<ResponseField name="timeout" type="Optional[float]">
  The timeout for requests in seconds.
</ResponseField>
<ResponseField name="host" type="Optional[str]">
  The host address of the Qdrant server.
</ResponseField>
<ResponseField name="path" type="Optional[str]">
  The path to the Qdrant database.
</ResponseField>
